// For Way A (JSON file) – copy loadGNMath exactly
async function loadMyCoolSource() {
  try {
    const resp = await fetch(sources.mysupercoolsource.url + "?t=" + Date.now());
    const json = await resp.json();
    sources.mysupercoolsource.games = json;
    displayZones(json);
  } catch (e) {
    document.getElementById('games-grid').innerHTML = `<p style="color:red;text-align:center;">My Cool Source failed to load.</p>`;
  }
}

// For Way B (static list like UGS) – copy initUGS exactly
function initMyCoolSource() {
  sources.mysupercoolsource.games = sources.mysupercoolsource.rawList.map(name => ({
    id: "mycool_" + name,
    name: name.replace(/_/g, ' ').replace(/\b\w/g, c => c.toUpperCase()),
    url: name + ".html",           // change .html if your files are .htm or whatever
    author: "My Cool Source",
    authorLink: "#"
  }));
  displayZones(sources.mysupercoolsource.games);
}
